const SCREENS = { MAP: 'map', COMBAT: 'combat', REWARD: 'reward', REST: 'rest', SHOP: 'shop', EVENT: 'event', GAMEOVER: 'gameover' };

function newGame() {
  return {
    deck: starterDeck.map(c => ({ ...c })),
    player: { hp: 80, maxHp: 80, gold: 100, block: 0, effects: {} },
    map: generateMap(15),
    currentNode: null,
    screen: SCREENS.MAP,
    combatState: null,
    rewardCards: null,
    shopCards: null,
    currentEvent: null,
    upgradeMode: false,
    gameResult: null,
    floorsCleared: 0,
  };
}

export default function App() {
  const [game, setGame] = useState(newGame);

  const selectNode = useCallback((floor, idx) => {
    setGame(g => {
      const node = g.map[floor][idx];
      const key = `${floor}-${idx}`;
      let next = { ...g, currentNode: key, floorsCleared: floor };

      switch (node.type) {
        case 'combat':
        case 'elite':
        case 'boss': {
          const enemies = getEnemiesForType(node.type, floor);
          const combatState = initCombat(next.deck, next.player, enemies);
          return { ...next, screen: SCREENS.COMBAT, combatState };
        }
        case 'rest':
          return { ...next, screen: SCREENS.REST, upgradeMode: false };
        case 'shop':
          return { ...next, screen: SCREENS.SHOP, shopCards: getShopCards(5) };
        case 'event':
          return { ...next, screen: SCREENS.EVENT, currentEvent: getRandomEvent() };
        default:
          return { ...next, screen: SCREENS.MAP };
      }
    });
  }, []);

  const onPlayCard = useCallback((card, targetIdx) => {
    setGame(g => {
      if (!g.combatState) return g;
      let cs = playCard(g.combatState, card, targetIdx);
      const result = isCombatOver(cs);
      if (result === 'won') {
        const rewards = getRandomRewardCards(3);
        const gold = 15 + Math.floor(Math.random() * 20);
        return {
          ...g,
          combatState: cs,
          player: { ...g.player, hp: cs.player.hp, gold: g.player.gold + gold },
          screen: SCREENS.REWARD,
          rewardCards: rewards
        };
      }
      if (result === 'lost') {
        return { ...g, combatState: cs, player: { ...g.player, hp: 0 }, screen: SCREENS.GAMEOVER, gameResult: 'lost' };
      }
      return { ...g, combatState: cs };
    });
